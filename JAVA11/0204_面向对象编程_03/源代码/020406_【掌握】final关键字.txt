class Ball {
	// 使用final定义的变量就成为了常量，常量在定义时必须赋值，
	// 并且不允许修改，常量全部大写
	final int START = 1 ;	// 启动状态
	final int STOP = 0 ;	// 停止状态
	public void change() {
		START = 0 ;	// 错误
	}
}
