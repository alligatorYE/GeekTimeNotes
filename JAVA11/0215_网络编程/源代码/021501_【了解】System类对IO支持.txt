package cn.mldn.demo;
import java.io.*;
public class TestDemo {
    public static void main(String args[]) throws Exception {
        InputStream input = System.in ; // 描述的是键盘输入
        byte data [] = new byte [1024] ; // 开辟一块空间
        System.out.print("请输入数据：" );
        int length = input.read(data) ; // 向字节数组进行读取
        System.out.println("【输入数据】" + new String(data, 0, length));
    }
}
