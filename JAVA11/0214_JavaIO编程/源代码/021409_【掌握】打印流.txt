package cn.mldn.demo;
import java.io.*;

class PrintUtil implements Closeable {   // 设计一个打印流的功能类，该类支持有各种数据类型输出流处理
    private OutputStream output ;
    public PrintUtil(OutputStream output) { // 设置输出的位置
        this.output = output ;
    }
    public void print(String str) {
        try {
            this.output.write(str.getBytes());
        } catch (IOException e) {
        }
    }
    public void print(int num) {
        this.print(String.valueOf(num));
    }
    public void println(String str) {
        this.println(str + "\r\n");
    }
    public void println(int num) {
        this.println(String.valueOf(num));
    }
    @Override
    public void close() throws IOException {
        this.output.close();
    }
}
public class TestDemo {
    public static void main(String args[]) throws Exception {
        PrintUtil pu = new PrintUtil(new FileOutputStream(new File("D:" + File.separator + "info.txt"))) ;
        pu.print("姓名：辛博") ;
        pu.print("、年龄：");
        pu.println(64);
        pu.close();
    }
}
