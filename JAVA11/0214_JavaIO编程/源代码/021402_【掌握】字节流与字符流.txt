package cn.mldn.demo;

import java.io.File;
import java.io.FileOutputStream;
import java.io.OutputStream;

public class TestDemo {
    public static void main(String args[]) throws Exception {
        // 1、定义要进行磁盘输出的完整路径
        File file = new File("D:" + File.separator + "mldn" + File.separator + "message" + File.separator + "info.txt");
        if (!file.getParentFile().exists()) {
            file.getParentFile().mkdirs() ; // 创建父路径
        }
        // 2、通过子类进行父抽象类的实例化
        OutputStream output = new FileOutputStream(file) ; // 获得了输出流
        // 3、实现数据的输出，数据的输出需要将内容变为字节数组
        output.write("www.mldn.cn".getBytes());
        output.close(); // 4、关闭输出流
    }
}

package cn.mldn.demo;

import java.io.*;

public class TestDemo {
    public static void main(String args[]) throws Exception {
        File file = new File("D:" + File.separator + "mldn" + File.separator + "message" + File.separator + "info.txt");
        if (file.exists()) {    // 文件路径存在
            InputStream input = new FileInputStream(file) ; // 实例化输入流对象
            byte data [] = new byte [1024] ; // 开辟一个空间进行数据读取
            int len = input.read(data) ; // 将输入流的内容读取到字节数组之中
            System.out.println("【" + new String(data, 0, len) + "】");
            input.close();
        }
    }
}
