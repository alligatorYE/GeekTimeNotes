class TicketThread implements Runnable {
    private volatile int ticket = 3 ;
    @Override
    public void run() {
        for (int x = 0 ; x < 100 ; x ++) {
            this.sale();
        }
    }
    private synchronized void sale() {
        if (this.ticket > 0) {
            try {
                Thread.sleep(100);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }

        }
    }
}
